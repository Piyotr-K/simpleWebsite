#!/bin/python3

from turtle import *
from random import randint

speed(0) # change the drawing speed

# goto(-150, -150)
# circle(180)
pu() # penup()

starting_x = -150
starting_y = 150
# Race course location
goto(starting_x, starting_y)

# pd() # pendown()
for step in range(11):
    write(step * 10, align = "center")
    right(90)
    forward(10)
    pd()
    for step in range(5):
    forward(20)
    pu()
    forward(20)
    pd()
    pu()
    backward(210) # Number of dashes * distance
    left(90)
    forward(30)

yuki = Turtle() # Creating a new pen
yuki.color('Green')
yuki.shape('turtle')
yuki.pu()

carol = Turtle()
carol.color('Blue')
carol.shape('turtle')
carol.pu()

ethan = Turtle()
ethan.color('red')
ethan.shape('turtle')
ethan.pu()

john = Turtle()
john.color('#FFBA00')
john.shape('turtle')
john.pu()

josh = Turtle()
josh.color('purple')
josh.shape('turtle')
josh.pu()

yuki.goto(starting_x, starting_y - 30)
carol.goto(starting_x, starting_y - (30 * 2))
ethan.goto(starting_x, starting_y - (30 * 3))
john.goto(starting_x, starting_y - (30 * 4))
josh.goto(starting_x, starting_y - (30 * 5))

for step in range(100):
    yuki.forward(randint(1, 5))
    carol.forward(randint(1, 5))
    ethan.forward(randint(1, 5))
