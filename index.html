    #!/bin/python3

    import turtle
    screen = turtle.Screen()
    
    image_urls = []
    
    with open("images.txt", "r") as f:
        for line in f:
        image_urls.append(line.strip())
    
    for url in image_urls:
        screen.addshape(url)
    
    for x in range(0, 10):
        print(x)
    
    melon = turtle.Turtle()
    melon.shape(image_urls[0])
    melon.speed(0)
    melon.left(90)
