import random as R

def randomInsult():
    word1 = R.choice(adj1)
    word2 = R.choice(adj2)
    word3 = R.choice(nouns)
    print("Thou art a(n) " + word1 + " " + word2 + " " + word3)

adj1 = []
adj2 = []
nouns = []

f = open("insults.csv")
# Alternatives to spliting by lines
# print(f.readlines())
# print(f.read().split("\n"))

for line in f.read().splitlines():
    pieces = line.split(",")
    adj1.append(pieces[0])
    adj2.append(pieces[1])
    nouns.append(pieces[2])

while True:
    randomInsult()
    user = input("Do you want another insult?")
    if user == "no":
        print("bye")
        break
