import guizero as GZ
import random as R

def something():
    global t, counter
    counter += 1
    t.value = f"Counter: {counter}"

    if counter >= 10:
        t.text_color = "Red"

def guess():
    global t, tb, rng
    user = tb.value
    user = int(user) # convert to integer

    if user == rng:
        t.value = "You Win!"
        t.text_color = "green"
    else:
        t.value = "Guess again!"
        t.text_color = "red"
        
app = GZ.App(title="Number Guesser")

rng = R.randint(1, 10)

# Text widget
t = GZ.Text(app, text="Guess the number!")
# Access properties of the Text() widget
t.font = "Comic Sans MS"

tb = GZ.TextBox(app, text="Type here", width=30)

# PushButton widget
b = GZ.PushButton(app, text="Guess", command=guess)
b.font = "Comic Sans MS"

app.display()
