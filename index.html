#!/bin/python3

def simpleSnowflake(x, y, size, color):
    snow.color(color)
    snow.pu()
    snow.setpos(x, y)
    snow.pd()
    for x in range(12):
        for x in range(2):
            snow.forward(size)
            snow.right(60)
            snow.forward(size)
            snow.right(120)
        snow.left(30)

def twig():
    snow.left(45)
    snow.forward(30)
    snow.backward(30)
    
    snow.right(45)
    snow.forward(30)
    snow.backward(30)
    
    snow.right(45)
    snow.forward(30)
    snow.backward(30)
    
    snow.left(45)

def branch():
    for i in range(numSegments):
        twig()
        snow.forward(twigLength)
    snow.backward(twigLength * numSegments)

def snowflake(x, y, color):
    snow.color(color)
    snow.pu()
    snow.setpos(x, y)
    snow.pd()
    numOfBranches = 8

    for i in range(numOfBranches):
        branch()
        snow.right(360 / numOfBranches)

def snowman(x, y, color):
    man.pu()
    man.setpos(x, y)
    man.pd()
    man.color(color)
    
    man.dot(100) # Creates a filled circle at your turtle's position
    
    # We do plus 50 here so we move the turtle pen upwards according to the y value entered
    man.setpos(x, y + 50)
