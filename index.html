Caesar 

#!/bin/python3

alphabet = "abcdefghijklmnopqrstuvwxyz"

file = open("message.txt", "r")
message = file.read().strip()

# word = input("Enter a message:")
key = input("Enter a key:")
key = int(key)

secret = ""
for letter in message:
    if letter == " ":
        secret += " "
    elif letter == "\n":
        secret += "\n"
    else:
        # finds the letter in the alphabet
        pos = alphabet.find(letter)
        
        # remainder division %
        # aka Modulus
        # 28 % 26 -> r: 2
        newpos = (pos + key) % 26
        
        # It builds a new word letter by letter
        secret += alphabet[newpos]

print("The secret message: " + secret)
print("Decoding...")

decoded = ""
for letter in secret:
    if letter == " ":
        decoded += " "
    elif letter == "\n":
        decoded += "\n"
    else:
        pos = alphabet.find(letter)
        newpos = (pos - key) % 26
        decoded += alphabet[newpos]
print("The decoded message: " + decoded)
