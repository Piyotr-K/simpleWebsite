import requests
from bs4 import BeautifulSoup

def parse_result(html):
    soup = BeautifulSoup(html, 'html.parser')
    images = soup.find_all('img')
    titles = soup.find_all('a', href=True)

    all_titles = []
    for title in titles:
        if ('items' in title['href']) and (title.string != None):
            all_titles.append(title.string)
    print(all_titles)

    # print(images)
    sources = []
    for src in images:
        buf = src['src']
        if ('amazon' not in buf) and ('login' not in buf):
            sources.append(buf)
    print(sources)

def request_site(url):
    try:
        response = requests.get(url)
        # Checking if our request has been accepted
        if response.status_code == 200:
            return response.text
    # If not then return None (null)
    except requests.RequestException:
        return None

Ciphers.py 

alphabet = 'abcdefghijklmnopqrstuvwxyz'
needMsg = True
needKey = True

while needMsg:
    message = input("Type a message:")
    if message.isnumeric():
        print("No numbers please")
    else:
        needMsg = False

while needKey:
    key = input("Type in a key (num):")
    if key.isnumeric():
        key = int(key)
        needKey = False
    else:
        print("No letters please")

final = ""

for char in message:
    if char == ' ':
        final += ' '
    else:
        # Find position of selected letter
        pos = alphabet.find(char)
    
        # Caesar cipher works by shifting the letter
        encoded = (pos + key) % len(alphabet)
    
        final += alphabet[encoded]

print("Encoded message: " + final)
