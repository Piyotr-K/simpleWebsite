#!/bin/python3

# str (String)
alphabet = "abcdefghijklmnopqrstuvwxyz"

# test = "hello there"
# Returns you the pos of the char in the str
# print(test.find("e")) # 1
# 17 % 6 -> 5 # modulus: take the remainder

msg = input("Enter a letter to be encrypted:")
key = input("Enter a key to encrypt with:")
key = int(key)

encoded = ""

for char in msg:
    pos = alphabet.find(char)
    pos = (pos + key) % 26
    encoded += alphabet[pos]

print("Encoded message:", encoded)

pos = alphabet.find(encoded)
pos = (pos - key) % 26
decoded = alphabet[pos]

print("Decoded message:", decoded)

#!/bin/python3

from random import randint

choice = input('''
    1 - Rock
    2 - Paper
    3 - Scissors
''')

choice = int(choice)

# Chooses a random number from low to high
cpu = randint(1, 3)

# Handle all ties
if choice == cpu:
    print("Tie")
elif choice == 1 and cpu == 2:
    print("CPU Wins!")
elif choice == 1 and cpu == 3:
    print("Player Wins!")
elif choice == 2 and cpu == 1:
    print("CPU Wins!")
elif choice == 2 and cpu == 3:
    print("Player Wins!")
elif choice == 3 and cpu == 2:
    print("CPU Wins!")
elif choice == 3 and cpu == 1:
    print("Player Wins!")
