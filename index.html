def template():
    out = ""
    out += "<!DOCTYPE html>\n"
    return out

def generate_head():
    out = ""
    out += "<head>\n"
    out += "<title>Great Books</title>\n"
    out += "</head>\n"
    return out

def write_to_file(string):
    f = open("books2.html", "w")
    f.write(string)
    f.close()

def main():
    final = ""
    final += template()
    final += generate_head()
    # generate_body()
    final += "\n</html>"
    write_to_file(final)

main()

import turtle

snow = turtle.Turtle()
turtle.Screen().bgcolor("gray")
snow.color("cyan")

# Set speed to max
snow.speed(0)
# Hide turtle
snow.ht()

twigLength = 30
numSegments = 4
numBranches = 8

def twig():
    snow.left(45)
    snow.forward(30)
    snow.backward(30)
    snow.right(45)
    snow.forward(30)
    snow.backward(30)
    snow.right(45)
    snow.forward(30)
    snow.backward(30)
    snow.left(45)

def branch():
    for i in range(numSegments):
    twig()
        snow.forward(twigLength)
    
        snow.backward(twigLength * numSegments)

def flake():
    # Snowflake Generation
    for i in range(numBranches):
        branch()
        snow.right(360 / numBranches)

snow.pu()
snow.goto(100, -100)
snow.pd()
flake()
snow.pu()
snow.goto(-100, 100)
snow.pd()
flake()
