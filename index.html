htmlWriter.py 

def template():
    out = ""
    out += "<!DOCTYPE html>\n"
    return out

def generate_head():
    out = ""
    out += "<head>\n"
    out += "<title>Great Books</title>\n"
    out += "</head>\n"
    return out

def generate_body(data):
    out = ""
    out += "<body>\n"
    out += "<ol>\n"
    out += generate_list(data)
    out += "</ol>\n"
    out += "</body>\n"
    return out

def generate_list(data):
    out = ""
    for line in data:
        out += "<li>"
        try:
            rank, title, author, image = line.split(",")
            out += title.strip() + ", "
            out += author.strip() + ", "
            out += image.strip()
        except ValueError:
            for item in line.split(",")[1:]:
                out += item.strip() + ", "
        except TypeError:
            continue
        out += "</li>\n"
    return out

def read_from_file(filename):
    f = open(filename, "r")
    out = f.readlines()
    f.close()
    return out

def write_to_file(string):
    f = open("books2.html", "w")
    f.write(string)
    f.close()

def main():
    text = read_from_file("top-books.txt")

    final = ""
    final += template()
    final += generate_head()
    final += generate_body(text)
    final += "\n</html>"
    write_to_file(final)

main()

# For loop in python
for step in range(11):
    # Code here will run up to the number in the range function times
    # Step starts at 0
    write(step * 10, align='center')
    right(90)
    forward(10)

    forward(30)
