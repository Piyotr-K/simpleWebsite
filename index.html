#!/bin/python3

def simpleSnowflake():
    for x in range(12):
        for x in range(2):
            snow.forward(100)
            snow.right(60)
            snow.forward(100)
            snow.right(120)
        snow.left(30)

def twig():
    snow.left(45)
    snow.forward(30)
    snow.backward(30)
    
    snow.right(45)
    snow.forward(30)
    snow.backward(30)
    
    snow.right(45)
    snow.forward(30)
    snow.backward(30)
    
    snow.left(45)

def branch():
    for i in range(numSegments):
        twig()
        snow.forward(twigLength)
    snow.backward(twigLength * numSegments)

def snowflake():
    numOfBranches = 8

    for i in range(numOfBranches):
        branch()
        snow.right(360 / numOfBranches)

import turtle
snow = turtle.Turtle()
tree = turtle.Turtle()

snow.ht()

twigLength = 30
numSegments = 4
