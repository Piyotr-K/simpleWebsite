# Try is used because some computers use the Tkinter package, but some have tkinter
try:
    import Tkinter, random, time, sys
except:
    import tkinter as Tkinter, sys, random, time

class Game(object):

    # ===============
    score = 0
    # ===============

    def __init__(self, canvas):
        self.canvas = canvas
        self.widthlimit = canvas.winfo_screenwidth()
        self.heightlimit = canvas.winfo_screenheight()
        # Reserving space for the 'bat'
        y2 = canvas.winfo_screenheight() - 70
        y1 = y2 - 15
        self.bat = Tkinter.Canvas.create_rectangle(canvas, 0, y1, 200, y2, fill='steelblue')
        self.ball = Tkinter.Canvas.create_oval(canvas, 0, 0, 50, 50, fill='red')
        self.xvelocity = 10
        self.yvelocity = 10
    
    def update(self, event=None):
        x1, y1, x2, y2 = self.canvas.coords(self.ball)
        bat_x1, bat_y1, bat_x2, bat_y2 = self.canvas.coords(self.bat)
        # Finds if any object is overlapping the bat object, aka checking for collision
        k = len(self.canvas.find_overlapping(bat_x1, bat_y1, bat_x2, bat_y2))

        # If the ball hits the leftmost of the window
        if x1 < 0:
            self.xvelocity = 2 # Change it so that it moves right
            self.canvas.move(self.ball, self.xvelocity, 0)
            pass
        # If the ball hits the bottom of the window
        elif y2 >= self.heightlimit:
            # Display Game Over Screen
            x = self.widthlimit / 2
            y = self.heightlimit / 2
            self.canvas.create_text(x, y, text='Game Over', font=('arial 50 bold'), fill='black')
            self.canvas.master.update()
            self.canvas.master.update_idletasks()
            time.sleep(2)
            self.canvas.master.destroy()
            sys.exit(0)
        # If the ball hits the rightmost of the window
        elif x2 >= self.widthlimit:
            self.xvelocity = -2
            self.canvas.move(self.ball, self.xvelocity, 0)
        # If the ball hits the top of the window
        elif y1 <= 0:
            self.yvelocity = 2
            self.canvas.move(self.ball, 0, self.yvelocity)
        # If the ball hits the bat
        elif k == 2:
            self.yvelocity = -2
            self.canvas.move(self.ball, 0, self.yvelocity)

            # =================================
            self.score += 1
            print(self.score)
            # =================================
        else:
            self.canvas.move(self.ball, self.xvelocity, self.yvelocity)
        pass
    
    # This is for controlling the bat with your keyboard
    def update_bat(self, event=None):
        direction = event.keysym
        x1, y1, x2, y2 = self.canvas.coords(self.bat)

        # If the bat is too far right
        if x2 > self.widthlimit:
            if direction == 'Right':
                self.canvas.move(self.bat, -20, 0)
            pass
        # If the bat is too far left
        elif x1 < 0:
            if direction == 'Left':
                self.canvas.move(self.bat, 20, 0)
            pass
        elif direction == 'Right':
            self.canvas.move(self.bat, 20, 0)
        elif direction == 'Left':
            self.canvas.move(self.bat, -20, 0)
        else:
            pass
