#!/bin/python3

bag = []

print("""
Welcome to your backpack

:)

What would you like to do with your bag?
1. View All Items
2. Add item
3. Delete item
4. Exit program
""")

while True:
    userChoice = int(input("Enter a number: "))
    # userChoice = int(userChoice)

    if userChoice == 1:
        print(bag)
    elif userChoice == 2:
        itemToAdd = input("Enter an item to add: ")
        bag.append(itemToAdd)
    elif userChoice == 3:
        print("Delete selected")
        bag.remove(0)
    elif userChoice == 4:
        break
    else:
        print("Invalid")

alphabet = "abcdefghijklmnopqrstuvwxyz" # alphabet as a string

f = open("sometext.txt", "r")

userIn = f.read()

encrypted_string = ""
for c in userIn:
    if c == " ":
        encrypted_string += " "
    else:
        originPos = alphabet.find(c)
        newPos = (originPos + 3) % 26 # transpose letter
        encrypted_string += alphabet[newPos]

print(encrypted_string)

# Don't have to actually create the file when using w
# it will create the file for you if it does not exist
f = open("encrypted_msg.txt", "w")
f.write(encrypted_string)
