import random
from guizero import App, Text, Picture, PushButton

def insult_me():
    adj1 = random.choice(col1)
    adj2 = random.choice(col2)
    noun = random.choice(col3)

    return f"Thou art a(n) {adj1} {adj2} {noun}"

def new_insult():
    new_roast = insult_me()
    text1.value = new_roast

col1 = []
col2 = []
col3 = []
with open("insults.csv", "r") as f:
    contents = f.read().splitlines()
    for line in contents:
        adj1, adj2, noun = line.split(",")
        col1.append(adj1)
        col2.append(adj2)
        col3.append(noun)

app = App("Insult Generator")
# Starting parameters is whenever you first create the new text
text1 = Text(app, insult_me())
btn1 = PushButton(app, new_insult, text="Insult Me")
btn2 = PushButton(app, None, text="Spice Me Up")
img1 = Picture(app, image = "chicken.jpg")

app.bg = "cyan"

text1.text_size = 16
text1.text_color = (255, 0, 0)

app.display()

from random import choice

players = []
teamNames = ['Team Badminton',
                'Team Softball',
                'Team Soccer',
                'Team Nice']
file = open('players.txt', 'r')
players = file.read().splitlines()
print(players)

newPlayer = input("Add a new player:")
players.append(newPlayer)
newPlayer = input("Add a new player:")
players.append(newPlayer)
newPlayer = input("Add a new player:")
players.append(newPlayer)

teamA = [] # creates a new "team"
teamB = []
teamC = []

while len(players) > 0:
    firstChoice = choice(players)
    teamA.append(firstChoice)
    players.remove(firstChoice)

    if players == []:
        break # If there are no more players; exit
    
    secondChoice = choice(players)
    teamB.append(secondChoice)
    players.remove(secondChoice)
    
    if players == []:
        break
    
    thirdChoice = choice(players)
    teamC.append(thirdChoice)
    players.remove(thirdChoice)

firstTeamName = choice(teamNames)
teamNames.remove(firstTeamName)
secondTeamName = choice(teamNames)
teamNames.remove(secondTeamName)
thirdTeamName = choice(teamNames)

print(firstTeamName, teamA)
print(secondTeamName, teamB)
print(thirdTeamName, teamC)
