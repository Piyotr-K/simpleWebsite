Main.py 

#!/bin/python3

from turtle import *
from random import randint

penup()
speed(0)
goto(-190, 190)

for step in range(11):
    write(step * 10, align="center")
    right(90)
    forward(10)
    pendown()
    forward(150)
    penup()
    backward(160)
    left(90)
    forward(30)

sam = Turtle()
sam.color("#87CEEB")
sam.shape("turtle")
sam.penup()
sam.goto(-190, 170)

michael = Turtle()
michael.color("green")
michael.shape("turtle")
michael.penup()
michael.goto(-190, 140)

for step in range(100):
    sam.forward(randint(1, 5))
    michael.forward(randint(1, 5))
    print(sam.xcor())
    print(michael.xcor())
