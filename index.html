Employee.py 

class Employee:

    # This is also another way of creating new class variables
    raise_amount = 1.04 # Float

    total_employees = 0

    # Constructor
    def __init__(self, first, last, pay):
        self.first = first
        self.last = last
        self.pay = pay
        self.email = f"{first}.{last}@companymail.com"
        Employee.total_employees += 1

    # So return is what you get out of using the method
    def fullname(self):
        return f"{self.first} {self.last}" # Here we return a string for using fullname

    def emailAndPay(self):
        return f"Pay: {self.pay}, Email: {self.email}"

    def applyraise(self):
        self.pay = int(self.pay * self.raise_amount)

    # cls means class
    @classmethod
    def set_raise_amount(cls, amount):
        cls.raise_amount = amount

    # ignores class and instance, there from the beginning
    @staticmethod
    def is_workday(day):
        if day.weekday() == 5 or day.weekday() == 6:
            return False
        return True

Wireless Communication

server.py 

import socket

# 127.0.0.1 localhost
localIP = "127.0.0.1"

# Port
localPORT = 20001

# Max size of message the server is ready to receive
bufferSize = 1024

# Create a socket datagram
udpSock = socket.socket(family=socket.AF_INET, type=socket.SOCK_DGRAM)

# Bind put ip and port to the socket
udpSock.bind((localIP, localPORT))

print("UDP Server up and running")

# Listen for datagrams
while True:
    # Listen for datagrams on the given socket
    dataAddress = udpSock.recvfrom(bufferSize)
    message = dataAddress[0].decode()

    print(f"Message from Client: {message}")

client.py 

import socket

msg = input("Type a message to the server: ")
bytesToSend = str.encode(msg)

serverAddressPort = ("127.0.0.1", 20001)
bufferSize = 1024

# Create a socket
udpSendSock = socket.socket(family=socket.AF_INET, type=socket.SOCK_DGRAM)

# Send to server using created UDP Socket
udpSendSock.sendto(bytesToSend, serverAddressPort)
