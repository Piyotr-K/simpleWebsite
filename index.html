#!/bin/python3

# str (String)
alphabet = "abcdefghijklmnopqrstuvwxyz"

# test = "hello there"
# Returns you the pos of the char in the str
# print(test.find("e")) # 1
# 17 % 6 -> 5 # modulus: take the remainder

msg = input("Enter a letter to be encrypted:")
key = input("Enter a key to encrypt with:")
key = int(key)

encoded = ""

for char in msg:
    pos = alphabet.find(char)
    pos = (pos + key) % 26
    encoded += alphabet[pos]

print("Encoded message:", encoded)

pos = alphabet.find(encoded)
pos = (pos - key) % 26
decoded = alphabet[pos]

print("Decoded message:", decoded)

    1 - Rock
    2 - Paper
    3 - Scissors
''')

choice = int(choice)

# Chooses a random number from low to high
