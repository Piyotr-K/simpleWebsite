Main.py 

from tkinter import *

spaces = 10

def Update():
    for x in range(0, spaces):
        print(" ", end="")
    print("*")

def moveRight():
    global spaces
    spaces += 1
    Update()

def moveLeft():
    global spaces
    if spaces < 1:
        return
    spaces -= 1
    Update()


def printRandom(event):
    print("Hello")

def leftClick(event):
    print("Left Click Detected!")

def rightClick(event):
    print("Right Click Detected!")

def leftArrow(event):
    moveLeft()

def rightArrow(event):
    moveRight()

def upArrow(event):
    print("^")

def downArrow(event):
    print("v")

root = Tk()

# window size
root.geometry("300x300")
root.title("Test")

root.bind("", leftClick)
root.bind("", rightClick)
root.bind("", leftArrow)
root.bind("", rightArrow)
root.bind("", upArrow)
root.bind("", downArrow)

Update()

root.mainloop()
