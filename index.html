snow = turtle.Turtle()

snow.ht()

snow.color("red")
turtle.Screen().bgcolor("black")

for x in range(12):
    for x in range(2):
        snow.forward(100)
        snow.right(60)
        snow.forward(100)
        snow.right(120)
