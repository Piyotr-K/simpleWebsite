# nicepassword123 -> uihfuiewbonuieobw323324hkjbh2j34k # encrypting/ciphering
# uihfuiewbonuieobw323324hkjbh2j34k -> nicepassword123  # decrypting/deciphering
# changing words into other words based on a pattern
# nicepassword123 -> plaintext
# uihfuiewbonuieobw323324hkjbh2j34k -> ciphertext

# key -> translates between ciphertext and plaintext

import turtle as T

def random():
    print("Hello there")

def moveUp():
    # Set heading (which direction its pointing in)
    player.seth(90)
    player.forward(10)

# headings: 0 (right), -90 (down), 180 (left)
screen = T.Screen()

player = T.Turtle()
player.color("red")
player.shape("turtle")
player.speed(0)
player.up()

# player.xcor() -> gives x position on the screen
# player.ycor() -> gives y position on the screen

screen.listen()
screen.onkey(random, "space")
screen.onkey(moveUp, "w")

T.mainloop()
