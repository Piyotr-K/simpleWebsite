#!/bin/python3

def insult_me(name):
    global list_a, list_b, list_c
    out = ""
    out += name + " is a "
    out += random.choice(list_a) + " "
    out += random.choice(list_b) + " "
    out += random.choice(list_c) + "."
    print(out)

import random

f = open('insults.txt', 'r')

list_a = []
list_b = []
list_c = []

for line in f.read().splitlines():
    # print(line.split(","))
    word1, word2, word3 = line.split(",")
    list_a.append(word1)
    list_b.append(word2)
    list_c.append(word3)

name = input("Enter a name:")
insult_me(name)
