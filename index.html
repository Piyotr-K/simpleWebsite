import random
from guizero import App, Text, PushButton, Combo

def insult_me():
    word_a = random.choice(list_a)
    word_b = random.choice(list_b)
    word_c = random.choice(list_c)
    insult = "Thou art a(n) " + word_a + " " + word_b + \
        " " + word_c

    return insult

def new_insult():
    new_insult = insult_me()
    message.value = new_insult

list_a = []
list_b = []
list_c = []
