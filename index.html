#!/bin/python3

import turtle
import random

colors = ["red", "orange", "yellow", "green", "blue", "indigo", "purple"]

snow = turtle.Turtle()

def moveFlake(x, y):
    snow.pu() # pen up first
    snow.goto(x, y) # move
    snow.pd() # pen dowN

def twig(size):
    snow.left(3 * size)
    snow.forward(2 * size)
    snow.backward(2 * size)
    snow.right(3 * size)
    snow.forward(2 * size)
    snow.backward(2 * size)
    snow.right(3 * size)
    snow.forward(2 * size)
    snow.backward(2 * size)
    snow.left(3 * size)

def branch(size):
    twigLength = 2 * size
    numOfSegments = 4
    
    for i in range(numOfSegments):
    twig(size)
    snow.forward(twigLength)
    
    snow.backward(twigLength * numOfSegments)

def snowflake(size):
    numOfBranches = 12

    for i in range(numOfBranches):
    branch(size)
    snow.right(30)

color_chosen = random.choice(colors)

turtle.Screen().bgcolor("black")
snow.color(color_chosen)
snow.speed(0)
snow.ht() # hide turtle

moveFlake(100, 100) # move the center point of the snowflake

snowflake(10)

color_chosen = random.choice(colors)
snow.color(color_chosen)
moveFlake(-100,-100)

snowflake(20)
