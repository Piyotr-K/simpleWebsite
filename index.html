from tkinter import *

def btnPress(num):
    global eq
    eq = eq + str(num)
    equation.set(eq)

root = Tk()

equation = StringVar()
calc = Label(root, textvariable = equation)

# Calculator panel
equation.set("23 + 45")
calc.grid(columnspan=4)

eq = ""

# Creating the buttons
Button0 = Button(root, text="0", command = lambda: btnPress(0), height="1", width="7")
Button1 = Button(root, text="1", command = lambda: btnPress(1))

Button0.grid(row=2, column=1)
Button1.grid(row=1, column=0)

root.mainloop()
