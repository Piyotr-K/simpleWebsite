#!/bin/python3

alphabet = 'abcdefghijklmnopqrstuvwxyz'
# a + 3 = d
# z + 3 = c
# prints 'h' because all lists in programming start at index 0
# print(alphabet[7])
# h = k
# e = h
# l = o
# o = r

user_key = input("Enter a key (enter a number): ")
if user_key.isnumeric():
    user_key = int(user_key)
    key = user_key
else:
    print("Invalid entry, using default key(3)")
    key = 3

done = False

while not done:
    message = input("Enter a word: ")
    if message.isnumeric():
    print("Invalid entry")
    else:
    done = True

encoded_string = '' # Define the string first

for letter in message:
    pos = alphabet.find(letter)
    
    if pos > (25 - key):
    pos = (pos % 25) + (key - 1)
    else:
    pos = (pos % 25) + key # 25 % 25
    
    encoded_string += alphabet[pos] # Concatenate (add) to the end of the string

print("Encoded Message: " + encoded_string)

print("Decoding Message...")

decoded_string = ''

for letter in encoded_string:
    
    pos = alphabet.find(letter)
    
    if pos < key:
    pos = (25 - pos) + (key - 1) # c = 2 -> z = 25
    else:
    pos = pos - key
    
    decoded_string += alphabet[pos]

print("Decoded Message: " + decoded_string)
