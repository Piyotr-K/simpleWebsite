class Player(object):

    # default attribute / variable
    health = 100
    number_of_players = 0
    age = 30

    # Constructor function
    def __init__(self, name, color):
        self.name = name # class attributes/variables
        self.color = color
        Player.number_of_players += 1

    # Print function
    def print_player(self):
        print('{} {}'.format(self.name, self.color))
    
    def apply_damage(self, damage):
        self.health -= damage
    
    @classmethod
    def printAge(cls):
        print("Age is: " + str(cls.age))

Enemy.py 

class Enemy:

    def __init__(self, name, faction):
        self.name = name
        self.faction = faction

    # Representation of the class
    # def __repr__(self):
    #     return "Enemy('{}', '{}')".format(self.name, self.faction)

    # Similar to repr, but for when you need to convert to string
    def __str__(self):
        return "Enemy('{}', '{}')".format(self.name, self.faction)
    
    def printData(self):
        print(self.name, self.faction, sep=" - ")
    
class Ghost(Enemy):

    # Override method (same name as parent)
    def printData(self):
        print("Ghost's method -> ", self.name, self.faction, sep=" - ")
