#!/bin/python3
import pygal

def readFile():
  out = []
  
  file = open('data.txt', 'r')

  for line in file.read().splitlines():
    if line:
      out.append(line)
  file.close()
  
  return out

def writeFile(data):
  file = open('data.txt', 'w')
  
  for line in data:
    file.write(line + "\n")
  file.close()
