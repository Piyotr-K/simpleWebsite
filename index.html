#!/bin/python3

import turtle

def simpleSnowflake(x, y, br):
    snow.pu()
    snow.goto(x, y)
    snow.pd()
    
    for i in range(br):
        for i in range(2):
            snow.forward(100)
            snow.right(60)
            snow.forward(100)
            snow.right(120)
        snow.right(360 / br)

def twig():
    snow.left(45)
    snow.forward(30)
    snow.backward(30)
    snow.right(45)
    snow.forward(30)
    snow.backward(30)
    snow.right(45)
    snow.forward(30)
    snow.backward(30)
    snow.left(45)

def branch():
    for i in range(numOfSegments):
        twig()
        snow.forward(twigLength)

    snow.backward(twigLength * numOfSegments)

def snowflake(x, y, br):
    snow.pu()
    snow.goto(x, y)
    snow.pd()

    for i in range(br):
        branch()
        snow.right(360 / br)

snow = turtle.Turtle()
snow.speed(0)
snow.color("cyan")
turtle.Screen().bgcolor("Dark Gray")
snow.ht()

twigLength = 30
numOfSegments = 4
numOfBranches = 6

snowflake(100, -100, 6)
snowflake(0, 200, 8)
