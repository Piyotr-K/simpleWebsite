Main.py 

from random import choice

players = []
# Read local files
file = open('names.txt', 'r')
players = file.read().splitlines()
print(players)

teamA = []
teamB = []
teamC = []

while len(players) > 0:
    selected = choice(players)
    teamA.append(selected) # Append = add to end of list
    players.remove(selected)
    
    if players == []:
    break # exits out of loop at this line

    selected = choice(players)
    teamB.append(selected)
    players.remove(selected)
    
    if players == []:
    break # exits out of loop at this line

    selected = choice(players)
    teamC.append(selected)
    players.remove(selected)

print("Team A:")
print(teamA)

print("Team B:")
print(teamB)

print("Team C:")
print(teamC)

f = open('Final.txt', 'w')
f.write('Something')
