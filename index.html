More Turtle 

#!/bin/python3

from turtle import *

screen = Screen()

speed(0)

# Make a new "pen"
pen = Turtle()
pen.speed(0)

# Changes the background colour
screen.bgcolor('gold')

# (font, size, font-style)
style = ('Comic Sans MS', 30, 'bold')
style2 = ('Arial', 20, 'italic')

pu()
setpos(0, 20)
# (message, style, alignment)
write('Piyotr Kao', font = style, align = 'center')

setpos(0, -20)
write('Teacher Mans', font = style2, align = 'center')

pen.pu()
# Set position(x, y)
pen.setpos(0, -135)
pen.pd()
# draw a circle
pen.circle(150)
pen.ht()

pen.color("red")
pen.pu()
pen.dot(60)

pen.goto(150, 15)
pen.dot(60)

pen.goto(0, 165)
pen.dot(60)

pen.goto(-150, 15)
pen.dot(60)

pen.setpos(0, -135)
pen.seth(45)
pen.goto()

# Hide the arrow thing
ht()
