#!/bin/python3

import pygal

f = open('data.txt', 'r')
allData = f.read().splitlines()

piechart = pygal.Pie()
piechart.title = 'Favourite Pets'

for data in allData:
    if data == '':
    break
    label, value = data.split(' ')
    piechart.add(label, int(value))

piechart.render()

while True:
    choice = input('''
    Pick your favourite movie:
    1 - Avengers Infinity War
    2 - Free Guy
    3 - Matrix
    4 - Kill Bill
    5 - Pulp Fiction
    6 - Shutter Island
    Press 'q' when done
    ''')
    print(choice)
    if choice == 'q':
        break

barGraph2 = pygal.Bar()

barGraph2.add('Matrix', 6)

barGraph2.add('Shutter Island', 9)
barGraph2.render()

listOfStuff = ["Money", "Toys", "Government", "Russia"]

# Print out the length of the list
print(len(listOfStuff))

# Print out all of the stuff in the list
print(listOfStuff)

# Print individual items
print(listOfStuff[0]) # REMEMBER PROGRAMMING STARTS COUNTING AT 0

# Change items in a list
listOfStuff[0] = "MoarMoney"
print("After changing the first item:", listOfStuff)

# Print out the last item of the list using len
print(listOfStuff[len(listOfStuff) - 1])
