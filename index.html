from tkinter import *
from random import *

def phrase_display():
    global entry, text
    greetings = ["Hi", "Hello", "S'up"]

    name = entry.get()

    greeting = choice(greetings) + " " + name

    text.insert(END, greeting)
    print(greeting)

window = Tk()

window.title("Test")
window.geometry("500x500")

lbl = Label(window, text="Username").grid()
label = Label(text="My Label").grid(row=1, column=0)

# Buttons
playButton = Button(window, text="Play", command=phrase_display).grid(row=1, column=1)

button = Button(window, text="Click Me")
button.grid()

# Entry and Textbox
entry = Entry()
entry.grid()

text = Text(window, height=10, width=30)
text.grid()

window.mainloop()
