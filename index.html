#!/bin/python3

import turtle

def test():
    print("Hello!")

def moveUp():
    # setheading - point the turtle in that direction
    playerT.seth(90)
    playerT.forward(20)

def moveDown():
    playerT.seth(270)
    playerT.forward(20)

def moveRight():
    playerT.seth(0)
    playerT.forward(20)

def moveLeft():
    playerT.seth(180)
    playerT.forward(20)

def trailToggle():
    global trail
    if not trail:
        trail = True
        playerT.pd()
    else:
        trail = False
        playerT.pu()

screen = turtle.Screen()
screen.bgcolor("beige")
screen.listen()

# param 1 - the function to run when you press that key
# param 2 - the key to listen for
# screen.onkey(test, "h")

playerT = turtle.Turtle()
playerT.shape("turtle")
playerT.color("red")
playerT.pu()
playerT.speed(0)
trail = False

screen.onkey(moveUp, "w")
screen.onkey(moveDown, "s")
screen.onkey(moveLeft, "a")
screen.onkey(moveRight, "d")
screen.onkey(trailToggle, "f")

#!/bin/python3

# list of stuff
# 1 - 1 toy
# 2 - 1 Money
# 3 - Own the government
listOfStuff = ["Money", "Toys", "Government", "Russia"]

# Print out the length of the list
print(len(listOfStuff))

# Print out all of the stuff in the list
print(listOfStuff)

# Print individual items
print(listOfStuff[0]) # REMEMBER PROGRAMMING STARTS COUNTING AT 0

# Change items in a list
listOfStuff[0] = "MoarMoney"
print("After changing the first item:", listOfStuff)

# Print out the last item of the list using len
print(listOfStuff[len(listOfStuff) - 1])

# Empty a list
listOfStuff = []

print("After erasing:", listOfStuff)

# Adding to the list
listOfStuff.append("Hee")
listOfStuff.append("Ho")

print("After adding to the list:", listOfStuff)

# Removing from a list with the name of the item
listOfStuff.remove("Hee")

# Removing from a list with the position number
listOfStuff.remove(listOfStuff[0])

print("After removing hee ho:", listOfStuff)
