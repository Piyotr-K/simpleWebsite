from tkinter import *
import random

def createRect(x1, y1, x2, y2, color):
    canvas.create_rectangle(x1, y1, x2, y2, fill=color)

def randomRects(num):
    colors = ["red", "green", "blue", "orange", "purple"]
    for i in range(num):
        color = random.choice(colors)
        x1 = random.randrange(150)
        y1 = random.randrange(150)
        x2 = random.randrange(150) + x1
        y2 = random.randrange(150) + y1
        createRect(x1, y1, x2, y2, color)
