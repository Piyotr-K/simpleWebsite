sheep.py 

from animal import Animal

class Sheep(Animal):

    def __init__(self, name, color, rate):
        super().__init__(name)
        self.color = color
        self.growth = rate
    
    def speak(self):
        return f"{self.name} the sheep says Baaa!"
    
    def wool(self):
        return f"{self.name} the sheep has {self.color} coloured wool."
    
    def rate(self):
        return f"{self.name} the sheep's {self.color} wool grows at {self.growth} days per full coat."
    
    def eat(self, food):
        if food == "Takoyaki":
            return f"{self.name} likes {food}"
        elif food == "Hay":
            return f"{self.name} likes {food}"
        return f"{self.name} dislikes {food}"

#!/bin/python3
from random import randint

player = input('''
Rock - 1
Paper - 2
Scissors - 3
''')

# Convert to int to compare
player = int(player)

# Randint = random integer
cpu = randint(1, 3) # 1 = r, 2 = p, 3 = s

if player == cpu:
    print("Tie")
elif player == 1 and cpu == 2:
    print("CPU Wins")
elif player == 1 and cpu == 3:
    print("Player Wins")
elif player == 2 and cpu == 1:
    print("Player Wins")
elif player == 2 and cpu == 3:
    print("CPU Wins")
elif player == 3 and cpu == 1:
    print("Player Wins")
elif player == 3 and cpu == 2:
    print("CPU Wins")
else:
    print("Not a choice")
