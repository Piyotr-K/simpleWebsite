#!/bin/python3

from random import randint

choice = input('''
    1 - Rock
    2 - Paper
    3 - Scissors
''')

choice = int(choice)

# Chooses a random number from low to high
cpu = randint(1, 3)

# Handle all ties
if choice == cpu:
    print("Tie")
elif choice == 1 and cpu == 2:
    print("CPU Wins!")
elif choice == 1 and cpu == 3:
    print("Player Wins!")
elif choice == 2 and cpu == 1:
    print("CPU Wins!")
elif choice == 2 and cpu == 3:
    print("Player Wins!")
elif choice == 3 and cpu == 2:
    print("CPU Wins!")
elif choice == 3 and cpu == 1:
    print("Player Wins!")
