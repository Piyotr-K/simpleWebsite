import pygal

piechart = pygal.Pie(inner_radius = 0.6)
piechart.title = 'Favourite Pets'
piechart.add('Dog', 60)
piechart.add('Cat', 20)
piechart.add('Fish', 10)
piechart.add('Frog', 3)
piechart.add('Chicken', 2)
piechart.render_to_file('piechart.svg')

bargraph = pygal.Bar()
bargraph._title = 'Pets part 2'

file = open('pets.txt', 'r')

for line in file.read().splitlines():
    if line:
        label, value = line.split(' ')
        bargraph.add(label, int(value))

file.close()

bargraph.render_to_file('bargraph.svg')

caesar.py 

#!/bin/python

alphabet = "abcdefghijklmnopqrstuvwxyz"
# message = input("Enter a message:")
f = open("sample.txt", "r")
message = f.read()

key = input("Enter a key:")
enc_msg = ""

# If the user didn't enter anything, set key to 3 by default
if not key:
    key = 3

for char in message:
    if char == " ":
        enc_msg += " "
    elif char == "\n":
        enc_msg += "\n"
    else:
        pos = alphabet.find(char)
        # Encoded char
        encoded = (pos + key) % 26
        # Encrypted message
        enc_msg += alphabet[encoded]

print("Encrypted message: " + enc_msg)
# Make another for loop and decrypt the message and print to console
dec_msg = ""

for char in enc_msg:
    if char == " ":
        dec_msg += " "
    elif char == "\n":
        dec_msg += "\n"
    else:
        pos = alphabet.find(char)
        decoded = (pos - key) % 26
        dec_msg += alphabet[decoded]

print("Decrypted message: " + dec_msg)
