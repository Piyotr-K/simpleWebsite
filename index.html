Employee.py 

class Employee:

    __fullname = ""
    __email = "@"

    def __init__(self, fullname="Default Name"):
        self.__fullname = fullname
        self.Email = fullname

    @property
    def Fullname(self):
        pass

    @Fullname.getter
    def Fullname(self):
        return self.__fullname

    @Fullname.setter
    def Fullname(self, value):
        self.__fullname = value
        self.Email = value

    @property
    def Email(self):
        pass

    @Email.getter
    def Email(self):
        return self.__email

    @Email.setter
    def Email(self, value):
        first, last = value.split(" ")
        self.__email = "{}.{}@mail.com".format(first, last)

    @Email.deleter
    def Email(self):
        self.__email = "Email Deleted"

from Employee import Employee

del emp1.Email
print(emp1.Email)
