Enemy.py 

class Enemy:
"""class description"""

def __init__(self, name, faction):
    self.name = name
    self.faction = faction

def printData(self):
    print(self.name, self.faction, sep="-")

def __str__(self): # To String method
    return "{}, {}".format(self.name, self.faction)

Ghost.py 

from Enemy import Enemy

class Ghost(Enemy):

    def printData(self):
        print("Ghost\'s method -->", self.name, self.faction, sep="-")

    def __add__(self, other):
        return self.faction + other.faction
