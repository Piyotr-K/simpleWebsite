snowflake.py 

#!/bin/python
import turtle

def twig():
    snow.left(45)
    snow.forward(30)
    snow.backward(30)
    snow.right(45)
    snow.forward(30)
    snow.backward(30)
    snow.right(45)
    snow.forward(30)
    snow.backward(30)
    snow.left(45)

def branch():
    for i in range(numOfSegments):
        twig()
        snow.forward(twigLength)
    snow.backward(twigLength * numOfSegments)

def snowflake(x, y, color):
    snow.pu()
    snow.goto(x, y)
    snow.pd()
    snow.color(color)
    for i in range(numBranches):
        branch()
        snow.right(360 / numBranches)

def snowflake2(x, y, color):
    snow.pu()
    snow.goto(x, y)
    snow.pd()
    snow.color(color)
    for i in range(10):
        for i in range(2):
            snow.forward(100)
            snow.right(60)
            snow.forward(100)
            snow.right(120)
        snow.right(36)

turtle.Screen().bgcolor("grey")

snow = turtle.Turtle()
snow.speed(0)
snow.ht()

twigLength = 30
numOfSegments = 4
numBranches = 4

snowflake(100, -100, "cyan")
snowflake(-100, 100, "gold")
snowflake2(100, 100, "purple")
