test.py 

import pygal

pie_chart = pygal.Pie(title="Favourite Pets")

file = open("data.txt", "r")

for line in file.read().splitlines():
    if line:
        label, value = line.split(' ') # Dog 6 -> ["Dog", "6"]
        print(f"Label: {label}, Value: {value}")
        pie_chart.add(label, int(value))

file.close()

pie_chart.render_in_browser() # To use this you need lxml -> pip install lxml
