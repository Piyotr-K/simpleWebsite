htmlWriter.py 

def template():
    out = ""
    out += "<!DOCTYPE html>\n"
    return out

def generate_head():
    out = ""
    out += "<head>\n"
    out += "<title>Great Books</title>\n"
    out += "</head>\n"
    return out

def generate_body(data):
    out = ""
    out += "<body>\n"
    out += "<ol>\n"
    out += generate_list(data)
    out += "</ol>\n"
    out += "</body>\n"
    return out

def generate_list(data):
    out = ""
    for line in data:
        out += "<li>"
        try:
            rank, title, author, image = line.split(",")
            out += title.strip() + ", "
            out += author.strip() + ", "
            out += image.strip()
        except ValueError:
            for item in line.split(",")[1:]:
                out += item.strip() + ", "
        except TypeError:
            continue
        out += "</li>\n"
    return out

def read_from_file(filename):
    f = open(filename, "r")
    out = f.readlines()
    f.close()
    return out

def write_to_file(string):
    f = open("books2.html", "w")
    f.write(string)
    f.close()

def main():
    text = read_from_file("top-books.txt")

    final = ""
    final += template()
    final += generate_head()
    final += generate_body(text)
    final += "\n</html>"
    write_to_file(final)

main()

#!/bin/python
from random import choice

# Homework:
# Make a team C, split the players accordingly write to final team
# Make 11 players in total
# Challenge:
# Allow the user at the beginning to enter more names
# Also allow user to enter file names (ex. names.txt, finalTeams.txt)

players = []
teamA = []
teamB = []

# open file and read from it
file = open('names.txt', 'r')
players = file.read().splitlines()
print(players)

# While there are still available players, split them into two teams
while len(players) > 0:
    # Pick the first player
    firstchoice = choice(players)
    # Add to first team
    teamA.append(firstchoice)
    # Remove player from available players
    players.remove(firstchoice)
    
    # Odd number of players
    # if all the players are chosen
    # exit the loop
    if players == []:
        break

    # Pick second player
    secondchoice = choice(players)
    # Add to second team
    teamB.append(secondchoice)
    # Remove player from available players
    players.remove(secondchoice)
print("Team A: " + str(teamA))
print("Team B: " + str(teamB))
print("Available Players: " + str(players))

# write to file
file2 = open("finalTeams.txt", "w")
file2.write("TeamA:" + "\n")
for member in teamA:
    file2.write(member + "\n")
file2.write("\nTeamB:" + "\n")
for member in teamB:
    file2.write(member + "\n")
file2.close()
