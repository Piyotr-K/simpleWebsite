#!/bin/python3
import turtle

def moveForward():
    global playerT
    playerT.forward(10)
    
def moveBackward():
    global playerT
    playerT.backward(10)
    
def moveLeft():
    global playerT
    playerT.left(90)

def moveRight():
    global playerT
    playerT.right(90)

screen = turtle.Screen()
screen.bgcolor("green")
screen.listen()

playerT = turtle.Turtle()
playerT.shape("turtle")
playerT.color("red")
playerT.speed(0)

finishT = turtle.Turtle()
finishT.ht()
finishT.color("white")
finishT.speed(0)

finishT.pu()
finishT.goto(180,-180)
finishT.pd()
finishT.seth(90)
finishT.forward(360)
finishT.write("FINISH", align="center")

screen.onkey(moveForward, "w")
screen.onkey(moveBackward, "s")
screen.onkey(moveLeft, "a")
screen.onkey(moveRight, "d")

while True:
    print("Player's current x: " +\
    str(playerT.xcor()) + " y: " +\
    str(playerT.ycor()))
