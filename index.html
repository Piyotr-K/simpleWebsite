#!/bin/python3

bag = []

print("""
Welcome to your backpack

:)

What would you like to do with your bag?
1. View All Items
2. Add item
3. Delete item
4. Exit program
""")

while True:
    userChoice = int(input("Enter a number: "))
    # userChoice = int(userChoice)

    if userChoice == 1:
        print(bag)
    elif userChoice == 2:
        itemToAdd = input("Enter an item to add: ")
        bag.append(itemToAdd)
    elif userChoice == 3:
        print("Delete selected")
        bag.remove(0)
    elif userChoice == 4:
        break
    else:
        print("Invalid")

import random as R

f = open("players.txt", "r")
players = f.read().splitlines()
print("Players reading done...")

teamA = []
teamB = []
teamC = []

while len(players) > 0:
    choice = R.choice(players)
    teamA.append(choice)
    players.remove(choice)
    
    if players == []:
        break

    choice = R.choice(players)
    teamB.append(choice)
    players.remove(choice)
    
    if players == []:
        break
    
    choice = R.choice(players)
    teamC.append(choice)
    players.remove(choice)
    
print("Team A: " + str(teamA))
print("Team B: " + str(teamB))
print("Team C: " + str(teamC))
