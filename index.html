#!/bin/python
import random

name = input("What is your name?")
gender = input("What is your gender?")

print("Hello " + name)

pronoun = ""

if gender == "girl":
    pronoun = "she"
elif gender == "boy":
    pronoun = "he"
else:
    pronoun = "it"

names = ["Leo", "Daisy", "Jonny", "Terence(MIA)"]
roles = ["Knight", "Princess", "Prince", "Wizard"]
places = ["Forest", "Castle", "Space Station"]
actions = ["Fight", "Kill", "Rescue"]

actor_name = random.choice(names)
actor_role = random.choice(roles)
setting = random.choice(places)
action = random.choice(actions)

story = "Once upon a time, there was a " + actor_role + " called " + name \
+ ". " + pronoun + " and some friends found themselves in the magic land of " \
+ setting + ". This land was ruled by " + actor_name + " the " + actor_role \
+ ". All of a sudden a mysterious voice spoke to them from high in the " \
+ "sky and said you must " + action + " " + actor_name + " the " + actor_role \
+ " to lift the curse of not being able to use technology..."

print(story)

enemy = "Goblin"

if setting == "Forest":
    print("You have entered the cursed forest of constipation")
elif setting == "Castle":
    print("You have entered the Castle of Hamsters")
elif setting == "Space Station":
    print("You have boarded the Alien :) Space Station")

print("You walked for 5 seconds and a " + enemy + " attacked you!")

defeated = False

player_hp = 100
monster_hp = 50

# Combat loop
while not defeated:
    print("Goblin's current hp: " + str(monster_hp))
    print("Your current hp: " + str(player_hp))
    print("What action do you take?")
    
    # Take Player action
    player = input("Attack (A) | Defend (D)")
    
    # Monster's action
    monster = random.choice(["a", "d"])
    
    # Test for Player's action and monster's actions
    player = player.lower()
    if player == "a":
        if monster == "a":
            player_hp -= 10
            monster_hp -= 15
        elif monster == "d":
            monster_hp -= 5

    # Check if monster's hp and Players' hps are below 0
    if monster_hp <= 0:
        print("You have defeated the " + enemy + "!")
        # Exit the loop through the variable
        defeated = True
    elif player_hp <= 0:
        print("You have died ;-;")
        print("Game Over")
        defeated = True

PlayerMovement.cs 

using System.Collections;
using System.Collections.Generic;
using UnityEngine;

public class PlayerMovement : MonoBehaviour
{

    private Rigidbody rb;
    // Start is called before the first frame update
    void Start()
    {
        rb = GetComponent<Rigidbody>();
    }

    // Update is called once per frame
    void Update()
    {
        float horizontal = Input.GetAxis("Horizontal");
        float vertical = Input.GetAxis("Vertical");

        horizontal *= 2;
        vertical *= 2;

        Vector3 move = new Vector3(horizontal, 0, vertical);

        move.y = rb.velocity.y;
        rb.velocity = move;
    }
}
